          ;

    {   
        char word[]="t";
        char *test=gencode(word,$1,'+',$3);
        $$=test;

    }
  | E '-' E 
    {
        char word[]="t";
        char *test=gencode(word,$1,'-',$3);
        $$=test;
    }
  | E '%' E 
    {
        char word[]="t";
        char *test=gencode(word,$1,'%',$3);
        $$=test;
    }
  | E '*' E 
    {
        char word[]="t";
        char test=gencode(word,$1,'',$3);
        $$=test;
    }
  | E '/' E 
    {
        char word[]="t";
        char *test=gencode(word,$1,'/',$3);
        $$=test;
    }
  | '(' E ')' 
    {
        $$=$2;
    }
  | NUM 
    {
        $$=$1;
    }
  ;
